Token Usage (CSS)

/* Consumer rebrands by overriding accent scale */
:root {
  --color-accent-9: oklch(0.55 0.19 360); /* Blooming Lotus */
  --color-accent-10: oklch(0.50 0.19 360); /* hover */
  --color-accent-11: oklch(0.43 0.14 360); /* text on light bg */
  --color-accent-3: oklch(0.93 0.035 360); /* subtle bg */
}

/* Dark mode: same hue, shifted L, reduced C */
.dark {
  --color-accent-9: oklch(0.64 0.16 360); /* lighter, desaturated */
  --color-accent-11: oklch(0.75 0.11 360); /* text on dark bg */
}

/* Components use semantic tokens only */
.btn-primary {
  background: var(--color-accent-9);
  color: var(--color-accent-fg);
  &:hover { background: var(--color-accent-10); }
}

/* Surfaces provide depth via lightness, not shadows */
.card {
  background: var(--color-surface-2);
  border: 1px solid var(--color-surface-border);
  color: var(--color-surface-fg);
}
